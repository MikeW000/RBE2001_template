37 void PIDMotor::loop() {
38  if (millis() - lastTimeRunPID > myPID.sampleRateMs) {
39  lastTimeRunPID = millis();
40  } else {
41  return;
42  }
43  // Perform linear/sinusoidal interpolation
44  float interpElapsed = (float) (millis() - startTime);
45  if (interpElapsed < duration && duration > 0) {
46  // linear elapsed duration
47  float unitDuration = interpElapsed / duration;
48  if (mode == SIN) {
49  // sinusoidal ramp up and ramp down
50  float sinPortion = (cos(-PI * unitDuration) / 2) + 0.5;
51  unitDuration = sinPortion;
52  }
53  float setpointDiff = endSetpoint - startSetpoint;
54  float newSetpoint = startSetpoint + (setpointDiff * unitDuration);
55  Setpoint = newSetpoint;
56  } else {
57  // If there is no interpoation to perform, set the setpoint to the end state
58  Setpoint = endSetpoint;
59  }
60  // Done Performing linear/sinusoidal interpolation
61  calcVel(); // ensure the velocity value is kept up to date
62  if (mode != VEL) {
63  // Position mode
64  Input = (float) getPosition();
65  //check for integral windup
66  bool thisErrPositive = (Setpoint - Input) > 0;
67  if (thisErrPositive != lastErrPositive) {
68  myPID.clearIntegralBuffer();
69  }
70  lastErrPositive = thisErrPositive;
71  // done with integral windup
72  // run the PID compute function
73  Output = myPID.calc(Setpoint, Input);
74  // use the unit vector function to set the output of the PID
75  setOutputUnitVector(Output);
76  } else {
77  velocityLoop();
78  }
79 }

106 void PIDMotor::setOutputUnitVector(float out) {
107  // Store the unit vector output set
108  Output = out;
109  // default to stop
110  hardwareOutput = getOutputStop();
111  if (Output == 0.0) {
112  // read the stop value from the subclass and set it to the hardware output
113  hardwareOutput = getOutputStop();
114  } else if (Output > 1.0) {
115  // read the max value from the subclass and set it to the hardware output
116  hardwareOutput = getOutputMax();
117  } else if (Output < -1.0) {
118  // read the min value from the subclass and set it to the hardware output
119  hardwareOutput = getOutputMin();
120  } else {
121  if (Output > 0)
122  // for positive signel, use the upper deadband to max
123  hardwareOutput = myFmap(Output, 0, 1,
124  getOutputStop() + getOutputMaxDeadbad(), getOutputMax());
125  else
126  // for negative signel, use the lower deadband to min
127  hardwareOutput = myFmap(Output, -1, 0, getOutputMin(),
128  getOutputStop() - getOutputMinDeadbad());
129  }
130  // with the hardwre value computed, send to sublcass output set
131  setOutput(hardwareOutput);

192 double PIDMotor::calcVel() {
193  //current positions
194  double curPos = getPosition();
195  //current time
196  curTime = millis();
197  //time change in ms from last call
198  timeInterval = curTime - prevTime;
199  if (timeInterval < 20)
200  return Vel; // If this is polled too fast, return the cached velocity
201  //encoder ticks since last call
202  movement = curPos - prevPos;
203  //encoder ticks to degrees
204  movement = movement / ticksToDeg();
205  //timeInterval in seconds
206  timeInterval = timeInterval / 1000;
207  //Velocity in degrees per seconds
208  Vel = movement / timeInterval;
209  //sets curent vals to previous
210  prevPos = curPos;
211  prevTime = curTime;
212  return Vel;
213 }
